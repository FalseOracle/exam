using NUnit.Framework;

[TestFixture]
public class ProgramTests
{
    [Test]
    public void Test_NormalCase()
    {
        int[] arr = { 2, 3, 5 };
        var result = Program.ProcessArray(arr);
        Assert.AreEqual(2, result.productEven);  // 2
        Assert.AreEqual(-2, result.differenceOdd); // 3 - 5 = -2
    }

    [Test]
    public void Test_NoEvenNumbers()
    {
        int[] arr = { 3, 5, 7 };
        var result = Program.ProcessArray(arr);
        Assert.AreEqual(1, result.productEven);  // баг: возможно ожидался 0
        Assert.AreEqual(-9, result.differenceOdd); // 3 - 5 - 7 = -9
    }

    [Test]
    public void Test_SingleOddNumber()
    {
        int[] arr = { 7 };
        var result = Program.ProcessArray(arr);
        Assert.AreEqual(1, result.productEven);  // нет чётных
        Assert.AreEqual(7, result.differenceOdd); // один элемент
    }
}
